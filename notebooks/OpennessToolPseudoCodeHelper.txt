function openness_tool(
    B,
    target_type,                 # INPUT: string, either "high" or "low".
                                 #        "high" maps bathymetric high features using Negative Openness (NO).
                                 #        "low" maps bathymetric low features using Positive Openness (PO).

    bathymetry_sign_convention,  # INPUT: string, either:
                                 #        "elevation_positive_up": larger raster values mean higher seabed.
                                 #        "depth_positive_down": larger raster values mean deeper water.
                                 #        This is critical because the algorithm assumes larger Z = higher seabed.

    radius_cells,                # INPUT: integer.
                                 #        Search radius R in raster cells.
                                 #        Controls the feature scale detected.
                                 #        Small R detects small features; large R detects broad features.

    directions,                  # INPUT: list of directions.
                                 #        Usually 8 compass directions:
                                 #        N, NE, E, SE, S, SW, W, NW.
                                 #        Used to cast radial search lines from each raster cell.

    cell_size,                   # INPUT: float.
                                 #        Raster cell size h in projected map units, e.g. metres.
                                 #        Used to convert cell offsets into real horizontal distances.

    C_large,                     # INPUT: float.
                                 #        First openness standard-deviation scale parameter.
                                 #        For highs, this corresponds to NO STD Scale Large.
                                 #        For lows, this corresponds to PO STD Scale Large.

    C_small,                     # INPUT: float.
                                 #        Second openness standard-deviation scale parameter.
                                 #        For highs, this corresponds to NO STD Scale Small.
                                 #        For lows, this corresponds to PO STD Scale Small.

    area_threshold               # INPUT: float.
                                 #        Minimum feature polygon area A_T in map units squared.
                                 #        Polygons smaller than this are removed.
):

    # ---------------------------------------------------------------------
    # Step 1: Standardise bathymetry
    # ---------------------------------------------------------------------
    # INPUT:
    #     B = bathymetry raster.
    #
    # ASSUMPTION:
    #     The openness maths must use a working raster Z where:
    #
    #         larger Z = physically higher seabed.
    #
    #     If B is depth-positive-down, for example:
    #         shallow = 30 m, deep = 100 m,
    #     then larger B means deeper, not higher, so we invert it:
    #
    #         Z = -B.
    #
    # OUTPUT:
    #     Z = working elevation raster with positive-up seabed convention.

    if bathymetry_sign_convention == "depth_positive_down":
        Z = -B
    else:
        Z = B


    # ---------------------------------------------------------------------
    # Step 2: Calculate Positive and Negative Openness
    # ---------------------------------------------------------------------
    # INPUT:
    #     Z = working elevation raster.
    #     radius_cells = R.
    #     directions = radial directions used for horizon search.
    #     cell_size = h.
    #
    # FOR EACH CELL p:
    #     For every direction k and sampled cell q along that ray:
    #
    #         gamma(k, q) = atan((Z[q] - Z[p]) / distance(p, q))
    #
    #     where:
    #         gamma is the elevation angle from centre cell p to surrounding cell q.
    #         positive gamma means q is higher than p.
    #         negative gamma means q is lower than p.
    #
    # POSITIVE OPENNESS:
    #     PO looks upward from the centre cell.
    #
    #         PO(p) = mean over directions of:
    #
    #             90° - max(gamma along that direction)
    #
    #     Small PO means surrounding terrain rises around p.
    #     GA-SaMMT uses small PO to identify bathymetric lows.
    #
    # NEGATIVE OPENNESS:
    #     NO looks downward from the centre cell.
    #
    #         NO(p) = mean over directions of:
    #
    #             90° + min(gamma along that direction)
    #
    #     Small NO means surrounding terrain falls away from p.
    #     GA-SaMMT uses small NO to identify bathymetric highs.
    #
    # OUTPUT:
    #     PO = Positive Openness raster, values in degrees, expected range 0° to 180°.
    #     NO = Negative Openness raster, values in degrees, expected range 0° to 180°.

    PO, NO = calculate_positive_and_negative_openness(
        Z,
        radius_cells,
        directions,
        cell_size
    )


    # ---------------------------------------------------------------------
    # Step 3: Select the correct openness raster and local extrema type
    # ---------------------------------------------------------------------
    # INPUT:
    #     target_type = "high" or "low".
    #     PO, NO = openness rasters.
    #     Z = working elevation raster.
    #
    # HIGH FEATURE LOGIC:
    #     Bathymetric highs are mapped from Negative Openness.
    #
    #         O = NO
    #
    #     Candidate high polygons must contain at least one local top.
    #     A local top is a local maximum in Z.
    #
    # LOW FEATURE LOGIC:
    #     Bathymetric lows are mapped from Positive Openness.
    #
    #         O = PO
    #
    #     Candidate low polygons must contain at least one local bottom.
    #     A local bottom is a local minimum in Z.
    #
    # OUTPUT:
    #     O = openness raster used for thresholding.
    #     Extrema = raster/vector/point set of local tops or local bottoms.

    if target_type == "high":
        O = NO
        Extrema = local_maxima(Z)
    else:
        O = PO
        Extrema = local_minima(Z)


    # ---------------------------------------------------------------------
    # Step 4: Calculate openness statistics
    # ---------------------------------------------------------------------
    # INPUT:
    #     O = selected openness raster.
    #
    # ASSUMPTION:
    #     Statistics are calculated only over finite valid cells.
    #     Nodata and infinite values must be ignored.
    #
    # EQUATIONS:
    #     mu    = mean(O)
    #     sigma = standard_deviation(O)
    #
    # OUTPUT:
    #     mu = mean openness value.
    #     sigma = openness standard deviation.

    mu = mean(O over finite valid cells)
    sigma = std(O over finite valid cells)


    # ---------------------------------------------------------------------
    # Step 5: Calculate two openness thresholds
    # ---------------------------------------------------------------------
    # INPUT:
    #     mu, sigma = openness statistics.
    #     C_large, C_small = standard-deviation scale parameters.
    #
    # EQUATIONS:
    #     T_large = mu - C_large * sigma
    #     T_small = mu - C_small * sigma
    #
    # INTERPRETATION:
    #     GA-SaMMT selects cells with unusually small openness.
    #
    #     For highs:
    #         O = NO, so small NO indicates a high candidate.
    #
    #     For lows:
    #         O = PO, so small PO indicates a low candidate.
    #
    # OUTPUT:
    #     T_large = first openness threshold.
    #     T_small = second openness threshold.

    T_large = mu - C_large * sigma
    T_small = mu - C_small * sigma


    # ---------------------------------------------------------------------
    # Step 6: Create two binary masks
    # ---------------------------------------------------------------------
    # INPUT:
    #     O = selected openness raster.
    #     T_large, T_small = openness thresholds.
    #
    # EQUATIONS:
    #     M_large[p] = 1 if O[p] < T_large, else 0
    #     M_small[p] = 1 if O[p] < T_small, else 0
    #
    # OUTPUT:
    #     M_large = binary raster from first threshold.
    #     M_small = binary raster from second threshold.

    M_large = O < T_large
    M_small = O < T_small


    # ---------------------------------------------------------------------
    # Step 7: Determine strict and loose masks
    # ---------------------------------------------------------------------
    # INPUT:
    #     M_large, M_small = two binary masks.
    #
    # REASON:
    #     The parameter names "Large" and "Small" come from GA-SaMMT,
    #     but unusual parameter values can make either mask stricter.
    #
    # DEFINITION:
    #     strict mask = mask with fewer selected cells.
    #     loose mask  = mask with more selected cells.
    #
    # OUTPUT:
    #     M_strict = smaller/more selective binary mask.
    #     M_loose = larger/more inclusive binary mask.

    if selected_area(M_large) <= selected_area(M_small):
        M_strict = M_large
        M_loose = M_small
    else:
        M_strict = M_small
        M_loose = M_large


    # ---------------------------------------------------------------------
    # Step 8: Polygonise both masks
    # ---------------------------------------------------------------------
    # INPUT:
    #     M_strict, M_loose = binary rasters.
    #
    # OPERATION:
    #     Convert connected regions of 1-valued cells into polygons.
    #
    # ASSUMPTION:
    #     Connectivity must be defined.
    #     Usually use 8-connectivity for raster morphology unless the
    #     implementation needs to match a specific GIS setting.
    #
    # OUTPUT:
    #     P_strict = polygon set from stricter mask.
    #     P_loose = polygon set from looser mask.

    P_strict = polygonise(M_strict)
    P_loose = polygonise(M_loose)


    # ---------------------------------------------------------------------
    # Step 9: Keep only polygons containing local extrema
    # ---------------------------------------------------------------------
    # INPUT:
    #     P_strict, P_loose = candidate polygon sets.
    #     Extrema = local tops for highs or local bottoms for lows.
    #
    # EQUATION:
    #     Keep polygon P if:
    #
    #         P intersects Extrema
    #
    #     For highs:
    #         P must contain at least one local maximum of Z.
    #
    #     For lows:
    #         P must contain at least one local minimum of Z.
    #
    # OUTPUT:
    #     Filtered P_strict and P_loose.

    P_strict = [P for P in P_strict if intersects(P, Extrema)]
    P_loose = [P for P in P_loose if intersects(P, Extrema)]


    # ---------------------------------------------------------------------
    # Step 10: Overlay split/merge logic
    # ---------------------------------------------------------------------
    # INPUT:
    #     P_strict = stricter candidate polygons.
    #     P_loose = looser candidate polygons.
    #
    # PURPOSE:
    #     Decide whether a broad candidate polygon should remain broad
    #     or be split into multiple stricter polygons.
    #
    # FOR EACH LOOSE POLYGON Q:
    #     Find all strict polygons P that are contained in Q or substantially
    #     inside Q.
    #
    # DECISION RULE:
    #     If Q contains more than one strict polygon:
    #         keep the multiple strict polygons.
    #
    #     If Q contains exactly one strict polygon:
    #         keep the broader loose polygon Q.
    #
    #     If Q contains no strict polygons:
    #         discard Q.
    #
    # MATHEMATICAL FORM:
    #     S(Q) = {P in P_strict : P is inside Q}
    #
    #     Output(Q) =
    #         S(Q),  if |S(Q)| > 1
    #         Q,     if |S(Q)| = 1
    #         none,  if |S(Q)| = 0
    #
    # OUTPUT:
    #     P_raw = raw merged/split polygon set before final cleaning.

    P_raw = []

    for Q in P_loose:

        contained = [
            P for P in P_strict
            if contained_or_substantially_inside(P, Q)
        ]

        if count(contained) > 1:
            P_raw.extend(contained)

        elif count(contained) == 1:
            P_raw.append(Q)

        else:
            continue


    # ---------------------------------------------------------------------
    # Step 11: Clean geometries and remove small features
    # ---------------------------------------------------------------------
    # INPUT:
    #     P_raw = raw candidate polygons.
    #     area_threshold = A_T.
    #
    # CLEANING:
    #     repair_geometries:
    #         Fix invalid polygon geometry.
    #
    #     remove_duplicates:
    #         Remove repeated polygons created by overlay operations.
    #
    # AREA FILTER EQUATION:
    #     Keep polygon P if:
    #
    #         area(P) >= A_T
    #
    # OUTPUT:
    #     P_final = final mapped bathymetric high or low feature polygons.

    P_raw = repair_geometries(P_raw)
    P_raw = remove_duplicates(P_raw)

    P_final = [
        P for P in P_raw
        if area(P) >= area_threshold
    ]


    # ---------------------------------------------------------------------
    # Step 12: Return outputs
    # ---------------------------------------------------------------------
    # OUTPUT:
    #     P_final = final polygon layer of mapped features.
    #
    #         If target_type == "high":
    #             P_final represents candidate bathymetric high features.
    #
    #         If target_type == "low":
    #             P_final represents candidate bathymetric low features.
    #
    #     O = openness raster used for mapping.
    #
    #         If target_type == "high":
    #             O = NO.
    #
    #         If target_type == "low":
    #             O = PO.

    return P_final, O
